Fetch API — Calling your C# API from HTML

Use fetch(), async/await, and JSON to talk to your Lesson 19 C# API live from this page.

How fetch() works in 3 steps

Step 1 — Make the request: const response = await fetch(url, options)

Step 2 — Check the result: if (!response.ok) throw new Error(...)

Step 3 — Read the JSON body: const data = await response.json()

Wrap everything in try { } catch(err) { } to handle network failures gracefully.

Example 1 — Load All Students

HTTP Method: GET  |  Endpoint: GET /api/students

Results will appear here…
// Step 1: mark the function async so we can use await
async function loadAllStudents() {
  try {
    // Step 2: fetch sends a GET request (default method)
    const response = await fetch('http://localhost:5000/api/students');

    // Step 3: if the server returned an error code, throw
    if (!response.ok) throw new Error(`HTTP error: ${response.status}`);

    // Step 4: parse the JSON body into a JS array
    const students = await response.json();

    // Step 5: display the data
    renderTable(students);
  } catch (err) {
    showError('output1', err.message);
  }
}

Example 2 — Find a Student by ID

HTTP Method: GET  |  Endpoint: GET /api/students/{id}

Enter an ID and click Find Student…
async function findStudent() {
  // Read the ID the user typed
  const id = document.getElementById('findId').value;
  if (!id) { showError('output2', 'Please enter an ID'); return; }

  try {
    // Template literals let us insert the id into the URL
    const response = await fetch(`http://localhost:5000/api/students/${id}`);

    if (response.status === 404) throw new Error(`No student with ID ${id}`);
    if (!response.ok) throw new Error(`HTTP error: ${response.status}`);

    const student = await response.json();
    renderTable([student], 'output2');
  } catch (err) {
    showError('output2', err.message);
  }
}

Example 3 — Add a New Student

HTTP Method: POST  |  Endpoint: POST /api/students

Fill in the form and click Add Student…
async function addStudent() {
  const name  = document.getElementById('newName').value.trim();
  const grade = document.getElementById('newGrade').value.trim();
  if (!name || !grade) { showError('output3', 'Name and grade are required'); return; }

  try {
    // POST needs extra options: method, headers, and a JSON body
    const response = await fetch('http://localhost:5000/api/students', {
      method:  'POST',
      headers: { 'Content-Type': 'application/json' },
      // JSON.stringify() converts a JS object into a JSON string
      body:    JSON.stringify({ name, grade })
    });

    if (!response.ok) throw new Error(`HTTP error: ${response.status}`);

    const created = await response.json();
    renderTable([created], 'output3');
  } catch (err) {
    showError('output3', err.message);
  }
}

Example 4 — Delete a Student

HTTP Method: DELETE  |  Endpoint: DELETE /api/students/{id}

Enter an ID and click Delete Student…
async function deleteStudent() {
  const id = document.getElementById('deleteId').value;
  if (!id) { showError('output4', 'Please enter an ID'); return; }

  try {
    // DELETE only needs the method changed — no body required
    const response = await fetch(`http://localhost:5000/api/students/${id}`, {
      method: 'DELETE'
    });

    if (response.status === 404) throw new Error(`No student with ID ${id}`);
    if (!response.ok) throw new Error(`HTTP error: ${response.status}`);

    // 204 No Content means success but no body to parse
    showSuccess('output4', `Student ID ${id} was deleted successfully.`);
  } catch (err) {
    showError('output4', err.message);
  }
}
